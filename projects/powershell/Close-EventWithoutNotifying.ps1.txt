function Execute-Command
{
[cmdletbinding()]
param( [string] $Exe = "hostname", [string] $Args = "" )

  # Setup the Process startup info
  $pinfo = New-Object System.Diagnostics.ProcessStartInfo
  $pinfo.FileName = "$Exe"
  if ($Args) {
      $pinfo.Arguments = " $Args"
  } else {
      $pinfo.Arguments = ""
  }
  $pinfo.UseShellExecute = $false
  $pinfo.CreateNoWindow = $true
  $pinfo.RedirectStandardOutput = $true
  $pinfo.RedirectStandardError = $true

  # For debug
  Write-Verbose -Message "           $Exe $Args"

  # Create a process object using the startup info
  $process = New-Object System.Diagnostics.Process
  $process.StartInfo = $pinfo

  # Start the process
  try {
    $rv = $process.Start() #| Out-Null
  } catch {
    Throw "$process.StandardError.ReadToEnd()"
  }

  # get output from stdout and stderr
  if ("$rv") {
     $stdout = "`n" + $process.StandardOutput.ReadToEnd() + $process.StandardError.ReadToEnd()
     return "$($stdout)"
  } else {
     # return null
  }
}  

function Update-Event
{
[cmdletbinding()]
param( [string] $EventID = "" )
	$sortableTimestamp = "{0:yyyy-MM-ddTHH:mm:ss}" -F $(Get-Date)
	Write-Verbose -Message "$sortableTimestamp Processing event $($EventID)"

	$SQLQUERY_SCRIPT = 'C:\ProgramData\HP\HPBTOS~1\bin\instrumentation\SQLQuery_PS_ForwardingInfo.ps1'
	
	# Annotate event
	C:\HPBSM\opr\bin\RestWsUtil.bat -create C:\ProgramData\HP\HPBTOS~1\bin\instrumentation\Add-Annotation.xml -verbose -username omirestuser -password p@55w0rd -url "https://opsbridge.unisuper.lan/opr-web/rest/event_list/$($EventID)/annotation_list"

	# Remove this event's forwarding record so it cannot notify Connected Servers on closure
	Remove-ForwardingInfo -EventID:$($EventID)

	# Close off event
	C:\HPBSM\opr\bin\RestWsUtil.bat -update C:\ProgramData\HP\HPBTOS~1\bin\instrumentation\Close-Event.xml -verbose -username omirestuser -password p@55w0rd -url "https://opsbridge.unisuper.lan/opr-web/rest/event_list/$($EventID)"

	# Postcheck
	$sqlQuery = "SELECT id,state FROM omi_event.dbo.all_events WHERE id = `'$($EventID)`'"
	C:\ProgramData\HP\HPBTOS~1\bin\instrumentation\SQLQuery_PS_ForwardingInfo.ps1 -sqlCommand:"$($sqlQuery)" -Datasource:"SQLAOAGUSMP1,1439"
}

function Remove-ForwardingInfo
{
[cmdletbinding()]
param( [string] $EventID = "" )

	# Remove the Forwarding Info from event   ######CAUTION: DISTRUCTIVE STATEMENT########
	$sqlStatement = "DELETE FROM omi_event.dbo.event_forwarding_info WHERE event_id like `'$($EventID)`' AND forwarding_type not like `'SYNCHRONIZE_AND_TRANSFER_CONTROL`'"
	Write-Host "                                  $($sqlStatement)"
	Write-Verbose -Message "           SQLQuery_PS_ForwardingInfo.ps1 -sqlCommand:`"$($sqlStatement)`" -Datasource:`"SQLAOAGUSMP1,1439`""
	C:\ProgramData\HP\HPBTOS~1\bin\instrumentation\SQLQuery_PS_ForwardingInfo.ps1 -sqlCommand:"$($sqlStatement)" -Datasource:"SQLAOAGUSMP1,1439"
}
